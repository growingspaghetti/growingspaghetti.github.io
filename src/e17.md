const ZERO_TO_19: [&str; 20] = [
    "",
    "one",
    "two",
    "three",
    "four",
    "five",
    "six",
    "seven",
    "eight",
    "nine",
    "ten",
    "eleven",
    "twelve",
    "thirteen",
    "fourteen",
    "fifteen",
    "sixteen",
    "seventeen",
    "eighteen",
    "nineteen",
];

const ZERO_TO_90: [&str; 10] = [
    "", "", "twenty", "thirty", "forty", "fifty", "sixty", "seventy", "eighty", "ninety",
];

fn count_words(i: usize) -> u32 {
    match i {
        0..=19 => ZERO_TO_19[i].len() as u32,
        20..=99 => {
            let p1 = i % 10;
            let p2 = i / 10;
            (ZERO_TO_90[p2].len() + ZERO_TO_19[p1].len()) as u32
        }
        _ if i % 100 == 0 && i != 1000 => {
            let d = i / 100;
            (ZERO_TO_19[d].len() + "hundred".len()) as u32
        }
        101..=999 => {
            let p3 = i / 100;
            let p2p1 = i - p3 * 100;
            (ZERO_TO_19[p3].len() + "hundredand".len()) as u32 + count_words(p2p1)
        }
        1000 => ("onethousand".len()) as u32,
        _ => 0,
    }
}

fn main() {
   let mut sum = 0u32;
    for i in 1..=1000 {
        sum += count_words(i);
    }

    println!("{}", sum);
    assert_eq!(sum, 21124);
}
