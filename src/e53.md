const ONE_MILLION: u32 = 1_000_000;

fn main() {
    let mut count = 0u32;
    let mut prev = [0u32; 101];
    let mut line = [0u32; 101];
    prev[0] = 1;
    for _ in 0..100 {
        line[0] = 1;
        for (i, &v) in prev.iter().enumerate() {
            if v == 0 {
                break;
            }
            let binom = v + prev[i + 1];
            line[i + 1] = if binom > ONE_MILLION {
                count += 1;
                ONE_MILLION
            } else {
                binom
            }
        }
        prev.copy_from_slice(&line);
    }

    println!("{}", count);
    assert_eq!(count, 4075);
}
