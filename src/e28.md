fn main() {
    let width = 1001u64;
    assert!(width % 2 == 1);
    let n = width / 2;
    let sum = 16 * n * (n + 1) * (2 * n + 1) / 6 + 4 * n * (1 + n) / 2 + 4 * n + 1;
    println!("{}", sum);
    assert_eq!(sum, 669171001);
}
