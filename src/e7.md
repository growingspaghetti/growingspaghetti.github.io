fn rule_out(sieve: &mut Vec<bool>, prime: usize) {
    for i in (prime..sieve.len()).step_by(prime) {
        sieve[i] = false;
    }
}

fn extend(sieve: &mut Vec<bool>, primes: &Vec<usize>) {
    sieve.extend(vec![true; sieve.len()]);
    for &p in primes {
        rule_out(&mut sieve.as_mut(), p);
    }
}

fn main() {
    let mut counter = 0u32;
    let mut sieve = vec![true; 10000];
    let mut primes: Vec<usize> = vec![];
    sieve[0] = false;
    sieve[1] = false;
    let mut cursor = 0usize;
    let n = 'label: loop {
        for i in cursor..sieve.len() {
            if !sieve[i] {
                continue;
            }
            counter += 1;
            if counter == 10001 {
                break 'label i as u64;
            }
            &primes.push(i);
            rule_out(&mut sieve, i);
        }
        cursor = sieve.len() - 1;
        extend(&mut sieve, &primes);
    };

    println!("{}", n);
    assert_eq!(n, 104743);
}

fn is_prime(n: u64) -> bool {
    if n < 2 {
        return false;
    }
    if n == 2 || n == 3 || n == 5 {
        return true;
    }
    for d in &[2u64, 3, 5] {
        if n % *d == 0 {
            return false;
        }
    }

    let side = (n as f64).sqrt() as u64;
    let mut d = 5u64;
    for i in [2u64, 4].iter().cycle() {
        if d > side {
            break;
        }
        d += *i;
        if n % d == 0 {
            return false;
        }
    }
    true
}

fn main() {
    let mut n = 0u64;
    let mut counter = 0u32;
    while counter < 10001 {
        n += 1;
        if is_prime(n) {
            counter += 1;
        }
    }
    
    println!("{}", n);
    assert_eq!(n, 104743);
}
