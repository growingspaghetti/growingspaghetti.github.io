- [Q.39 Integer right triangles](./e39.md)

```rust,editable
fn main() {
    let mut ans = None;
    'exploration: for m in 2..=(499f32.sqrt() as u32) {
        for n in 1..m {
            let a = m * m - n * n;
            let b = 2 * m * n;
            let c = m * m + n * n;
            if a + b + c == 1000 {
                ans = Some(a * b * c);
                break 'exploration;
            }
        }
    }

    println!("{:?}", ans);
    assert_eq!(ans.unwrap(), 31875000);
}
```
